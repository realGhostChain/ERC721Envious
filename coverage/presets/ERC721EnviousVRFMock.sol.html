pragma solidity ^0.8.0;
 
import "../extension/ERC721Envious.sol";
import "../interfaces/IBondDepository.sol";
import "../interfaces/INoteKeeper.sol";
import "../openzeppelin/token/ERC20/IERC20.sol";
import "../openzeppelin/token/ERC20/utils/SafeERC20.sol";
import "../openzeppelin/access/Ownable.sol";
import "../openzeppelin/utils/Counters.sol";
 
import "../chainlink/contracts/src/v0.8/interfaces/VRFCoordinatorV2Interface.sol";
import "../chainlink/contracts/src/v0.8/VRFConsumerBaseV2.sol";
 
/**
 * @title ERC721 Collateralization Royaly Mock
 * This mock shows an implementation of ERC721Envious with additional functionality
 * from ChainLink. Verified Random Function (VRF) will be used in order to
 * achieve on-chain randomness for differen types of "random collateralization".
 *
 * @author 5Tr3TcH @ghostown
 */
contract ERC721EnviousVEFMock is Ownable, ERC721Envious, VRFConsumerBaseV2 {
 
	using SafeERC20 for IERC20;
	using Counters for Counters.Counter;
 
	// addresses of GHST token and GhostBonding
	address private _ghostAddress;
	address private _ghostBondingAddress;
 
	// solhint-disable-next-line
	string private constant ZERO_ADDRESS = "zero address found";
	
	Counters.Counter private _nextTokenId;
 
	// ChainLink state variables below
 
	// Coordinator
	// see https://docs.chain.link/docs/vrf-contracts/#configurations
	address public vrfCoordinator;
	// Your subscription ID.
	uint64 public sSubscriptionId;
	// The gas lane to use, which specifies the maximum gas price to bump to.
	// For a list of available gas lanes on each network,
	// see https://docs.chain.link/docs/vrf-contracts/#configurations
	bytes32 public sKeyHash;
	// Depends on the number of requested values that you want sent to the
	// fulfillRandomWords() function. Storing each word costs about 20,000 gas,
	// so 40,000 is a safe default for this example contract. Test and adjust
	// this limit based on the network that you select, the size of the request,
	// and the processing of the callback request in the fulfillRandomWords()
	// function.
	uint32 public callbackGasLimit;
	// For this example, retrieve 1 random value in one request.
	// Cannot exceed VRFCoordinatorV2.MAX_NUM_WORDS.
	uint32 public numWords = 1;
	// The default is 3, but you can set this higher.
	uint16 public requestConfirmations;
 
	// map donator addresses to requestIds
	mapping(uint256 => address) private _donators;
	// map full amount to donator
	mapping(address => uint256) private _requestFullAmount;
	// map vrf results to donators
	mapping(address => uint256[]) private _requestRandomIds;
 
 
	constructor(
		string memory name,
		string memory symbol,
		address ghostAddress,
		address ghostBondingAddress,
		address token
	) ERC721(name, symbol) VRFConsumerBaseV2(vrfCoordinator) {
		require(
			ghostAddress != address(0) &&
			ghostBondingAddress != address(0) &&
			token != address(0),
			ZERO_ADDRESS
		);
 
		_changeGhostAddresses(ghostAddress, ghostBondingAddress);
		_changeCommunityToken(token);
		
		// nextTokenId is initialized to 1, since starting at 0 leads to higher gas cost 
		// for the first minter
		_nextTokenId.increment();
	}
 
	function getAmount(uint256 amount, address tokenAddress) external view returns (uint256) {
		uint256 circulatingSupply =
			IERC20(communityToken).totalSupply() - IERC20(communityToken).balanceOf(address(0));
		return amount * _scaledAmount(tokenAddress) / circulatingSupply;
	}
	
	function totalSupply() public view returns (uint256) {
		return _nextTokenId.current() - 1;
	}
 
	function initializeVRF(
		address _vrfCoordinator,
		uint64 _sSubscriptionId,
		bytes32 _sKeyHash,
		uint32 _callbackGasLimit,
		uint16 _requestConfirmations
	) external onlyOwner {
		vrfCoordinator = _vrfCoordinator;
		sSubscriptionId = _sSubscriptionId;
		sKeyHash = _sKeyHash;
		callbackGasLimit = _callbackGasLimit;
		requestConfirmations = _requestConfirmations;
	}
 
	function getDiscountedCollateral(
		uint256 bondId,
		address quoteToken,
		uint256 tokenId,
		uint256 amount,
		uint256 maxPrice
	) external {
		_addDiscountedCollateral(bondId, quoteToken, tokenId, amount, maxPrice);
	}
 
	function claimDiscountedCollateral(uint256 tokenId, uint256[] memory indexes) external {
		_claimDiscountedCollateral(tokenId, indexes);
	}
 
	function prepareRandomness() external returns (uint256 requestId) {
		require(_requestRandomIds[_msgSender()].length > 0, "vrf already used");
		requestId = VRFCoordinatorV2Interface(vrfCoordinator).requestRandomWords(
			sKeyHash,
			sSubscriptionId,
			requestConfirmations,
			callbackGasLimit,
			numWords
		);
 
		_donators[requestId] = _msgSender();
		_requestRandomIds[_msgSender()] = new uint256[](0);
		_requestFullAmount[_msgSender()] = type(uint256).max;
	}
	
	function collateralRandomTokens(
		uint256[] memory amounts, 
		address[] memory tokenAddresses
	) external payable {
		require(amounts.length == tokenAddresses.length, "lengths not match");
 
		uint256[] memory tokenIds = _requestRandomIds[_msgSender()];
		uint256 length = amounts.length > tokenIds.length ? tokenIds.length : amounts.length;
		uint256 ethAmount = msg.value;
		
		for (uint256 i = 0; i < length; i++) {
			if (tokenAddresses[i] == address(0)) {
				ethAmount -= amounts[i];
			}
			_addTokenCollateral(tokenIds[i], amounts[i], tokenAddresses[i]);
		}
		
		_nullifyRandomness(_msgSender());
	}
	
	function collateralRandomAmounts(
		uint256[] memory tokenIds, 
		uint256 amount, 
		address tokenAddress
	) external payable {
		if (tokenAddress == address(0)) {
			require(amount == msg.value, "amounts not match");
		}
 
		uint256[] memory tokenAllocations = _requestRandomIds[_msgSender()];
		uint256 fullAmount = _requestFullAmount[_msgSender()];
		uint256 length =
			tokenAllocations.length > tokenIds.length ? tokenIds.length : tokenAllocations.length;
		
		for (uint256 i = 0; i < length; i++) {
			uint256 tmpAmount = amount * tokenAllocations[i] / fullAmount;
			
			if (amount < tmpAmount) {
				_addTokenCollateral(tokenIds[i], amount, tokenAddress);
				break;
			}
			
			amount -= tmpAmount;
			_addTokenCollateral(tokenIds[i], tmpAmount, tokenAddress);
		}
		
		_nullifyRandomness(_msgSender());
	}
	
	function mint(address to) external onlyOwner {
		uint256 currentTokenId = _nextTokenId.current();
		_nextTokenId.increment();
		_safeMint(to, currentTokenId);
	}
 
	fallback() external payable {
		_disperseTokenCollateral(msg.value, address(0));
	}
 
	receive() external payable {
		_disperseTokenCollateral(msg.value, address(0));
	}
 
	function _changeGhostAddresses(address ghostToken, address ghostBonding) internal override {
		_ghostAddress = ghostToken;
		_ghostBondingAddress = ghostBonding;
	}
 
	function _addDiscountedCollateral(
		uint256 bondId,
		address quoteToken,
		uint256 tokenId,
		uint256 amount,
		uint256 maxPrice
	) internal override {
		// NOTE: this contract is temporary holder of `quoteToken` due to the need of
		// registration of bond inside. `amount` of `quoteToken`s should be empty in
		// the end of transaction.
		IERC20(quoteToken).safeTransferFrom(_msgSender(), address(this), amount);
		IERC20(quoteToken).safeApprove(_ghostBondingAddress, amount);
 
		(uint256 payout,, uint256 index) = IBondDepository(_ghostBondingAddress).deposit(
			bondId,
			amount,
			maxPrice,
			address(this),
			address(this)
		);
 
		if (payout > 0) {
			bondPayouts[tokenId] += payout;
			bondIndexes[tokenId].push(index);
		}
	}
 
	function _claimDiscountedCollateral(uint256 tokenId, uint256[] memory indexes) internal override {
		uint256 payout = INoteKeeper(_ghostBondingAddress).redeem(address(this), indexes, true);
 
		if (payout > 0) {
			(bool shouldAppend,) = _arrayContains(_ghostAddress, collateralTokens[tokenId]);
			if (shouldAppend) {
				collateralTokens[tokenId].push(_ghostAddress);
			}
			collateralBalances[tokenId][_ghostAddress] += payout;
			bondPayouts[tokenId] -= payout;
		}
 
		for (uint256 i = 0; i < indexes.length; i++) {
			uint256 index = _arrayContains(indexes[i], bondIndexes[tokenId]);
			bondIndexes[tokenId][index] = bondIndexes[tokenId][bondIndexes[tokenId].length - 1];
			bondIndexes[tokenId].pop();
		}
	}
 
	function _disperse(address tokenAddress, uint256 tokenId) internal override {
		uint256 balance = disperseBalance[tokenAddress] / totalSupply();
		if (balance > disperseTaken[tokenId][tokenAddress]) {
			uint256 amount = balance - disperseTaken[tokenId][tokenAddress];
			disperseTaken[tokenId][tokenAddress] += amount;
 
			(bool shouldAppend,) = _arrayContains(tokenAddress, collateralTokens[tokenId]);
			if (shouldAppend) {
				collateralTokens[tokenId].push(tokenAddress);
			}
			collateralBalances[tokenId][tokenAddress] += amount;
		}
	}
 
	// solhint-disable-next-line
	function fulfillRandomWords(uint256 requestId, uint256[] memory randomWords) internal override {
		// solhint-disable-next-line
		uint256 salt = block.difficulty + block.number + block.timestamp;
		uint256 forTime = uint256(keccak256(abi.encode(randomWords[0], salt)));
 
		uint256 total = totalSupply();
		uint256 fullAmount = 0;
		uint256 maxRandonNumbers = (forTime % totalSupply()) + 1;
 
		uint256[] memory expandedValues = new uint256[](maxRandonNumbers);
		for (uint256 i = 0; i < maxRandonNumbers; i++) {
			uint256 one = uint256(keccak256(abi.encode(randomWords[0], i)));
			uint256 two = uint256(keccak256(abi.encode(block.number, i)));
			uint256 result = one > two ? (one % two) : two % one;
			expandedValues[i] = (result % total) + 1;
			fullAmount += expandedValues[i];
		}
 
		_requestRandomIds[_donators[requestId]] = expandedValues;
		_requestFullAmount[_donators[requestId]]= fullAmount;
	}
 
	function _nullifyRandomness(address who) internal {
		_requestRandomIds[who] = new uint256[](0);
		_requestFullAmount[who] = type(uint256).max;
	}
}
