pragma solidity ^0.8.0;
 
import "../extension/ERC721Envious.sol";
import "../interfaces/IBondDepository.sol";
import "../interfaces/INoteKeeper.sol";
import "../openzeppelin/token/ERC721/extensions/ERC721Enumerable.sol";
import "../openzeppelin/token/ERC721/extensions/ERC721Burnable.sol";
import "../openzeppelin/token/ERC721/extensions/ERC721Pausable.sol";
import "../openzeppelin/token/ERC20/IERC20.sol";
import "../openzeppelin/token/ERC20/utils/SafeERC20.sol";
import "../openzeppelin/access/AccessControlEnumerable.sol";
import "../openzeppelin/utils/Counters.sol";
 
/**
 * @title ERC721 Collateralization Mock
 * This mock shows a generic implementation of ERC721Envious with additional
 * ability to use {IBondDepository} and {INoteKeeper}. ERC721Enumerable is used,
 * so in this example {_disperse} function is based on {totalSupply} of it.
 * While current smart contract will implement 5 extensions, resulting size is
 * ~23.59 KiB according to 1337 optimization runs.
 *
 * @author 5Tr3TcH @ghostown
 */
contract ERC721EnviousPreset is
	AccessControlEnumerable, 
	ERC721Burnable, 
	ERC721Pausable, 
	ERC721Enumerable,
	ERC721Envious
{
 
	using SafeERC20 for IERC20;
	using Counters for Counters.Counter;
 
	// addresses of GHST token and GhostBonding
	address private _ghostAddress;
	address private _ghostBondingAddress;
 
	// solhint-disable-next-line
	string private constant ONLY_OWNER = "incorrect role";
	// solhint-disable-next-line
	string private constant ZERO_ADDRESS = "zero address";
 
	bytes32 private constant MINTER_ROLE = keccak256("MINTER_ROLE");
	bytes32 private constant PAUSER_ROLE = keccak256("PAUSER_ROLE");
	bytes32 private constant GHOSTY_ROLE = keccak256("GHOSTY_ROLE");
 
	Counters.Counter private _tokenIdTracker;
 
	constructor(
		string memory name,
		string memory symbol,
		address ghostAddress,
		address ghostBondingAddress,
		address token
	) ERC721(name, symbol) {
		Erequire(
			ghostAddress != address(0) &&
			ghostBondingAddress != address(0) &&
			token != address(0),
			ZERO_ADDRESS
		);
 
		_changeGhostAddresses(ghostAddress, ghostBondingAddress);
		_changeCommunityToken(token);
 
		_setupRole(DEFAULT_ADMIN_ROLE, _msgSender());
 
		_setupRole(MINTER_ROLE, _msgSender());
		_setupRole(PAUSER_ROLE, _msgSender());
		_setupRole(GHOSTY_ROLE, _msgSender());
	}
 
	function supportsInterface(bytes4 interfaceId)
		public 
		view 
		override(AccessControlEnumerable, ERC721Envious, ERC721Enumerable, ERC721) 
		returns (bool) 
	{
		return ERC721Envious.supportsInterface(interfaceId) || 
			AccessControlEnumerable.supportsInterface(interfaceId) || 
			ERC721Enumerable.supportsInterface(interfaceId);
	}
 
	function addDiscountedCollateral(
		uint256 bondId,
		address quoteToken,
		uint256 tokenId,
		uint256 amount,
		uint256 maxPrice
	) external {
		_addDiscountedCollateral(bondId, quoteToken, tokenId, amount, maxPrice);
	}
 
	function mint(address to) external {
		require(hasRole(MINTER_ROLE, _msgSender()), ONLY_OWNER);
		_tokenIdTracker.increment();
		_mint(to, _tokenIdTracker.current());
	}
 
	function safeMint(address to) external {
		require(hasRole(MINTER_ROLE, _msgSender()), ONLY_OWNER);
		_tokenIdTracker.increment();
		_safeMint(to, _tokenIdTracker.current());
	}
 
	function safeMint(address to, bytes memory data) external {
		require(hasRole(MINTER_ROLE, _msgSender()), ONLY_OWNER);
		_tokenIdTracker.increment();
		_safeMint(to, _tokenIdTracker.current(), data);
	}
 
	function pause() external {
		require(hasRole(PAUSER_ROLE, _msgSender()), ONLY_OWNER);
		_pause();
	}
 
	function unpause() external {
		require(hasRole(PAUSER_ROLE, _msgSender()), ONLY_OWNER);
		_unpause();
	}
 
	function claimDiscountedCollateral(uint256 tokenId, uint256[] memory indexes) external {
		_claimDiscountedCollateral(tokenId, indexes);
	}
 
	function changeGhostAddresses(address ghostToken, address ghostBondingAddress) external {
		require(
			_ghostAddress != address(0) && _ghostBondingAddress != address(0),
			ZERO_ADDRESS
		);
		require(hasRole(GHOSTY_ROLE, _msgSender()), ONLY_OWNER);
		_changeGhostAddresses(ghostToken, ghostBondingAddress);
	}
 
	function changeCommunityToken(address newTokenAddress) external {
		require(newTokenAddress != address(0), ZERO_ADDRESS);
		require(hasRole(GHOSTY_ROLE, _msgSender()), ONLY_OWNER);
		_changeCommunityToken(newTokenAddress);
	}
 
	function changeComissions(uint256 incoming, uint256 outcoming) external {
		require(hasRole(GHOSTY_ROLE, _msgSender()), ONLY_OWNER);
		_changeComissions(incoming, outcoming);
	}
 
	fallback() external payable {
		_disperseTokenCollateral(msg.value, address(0));
	}
 
	receive() external payable {
		_disperseTokenCollateral(msg.value, address(0));
	}
 
	function _changeGhostAddresses(address ghostToken, address ghostBonding) internal override {
		_ghostAddress = ghostToken;
		_ghostBondingAddress = ghostBonding;
	}
 
	function _addDiscountedCollateral(
		uint256 bondId,
		address quoteToken,
		uint256 tokenId,
		uint256 amount,
		uint256 maxPrice
	) internal override {
		// NOTE: this contract is temporary holder of `quoteToken` due to the need of
		// registration of bond inside. `amount` of `quoteToken`s should be empty in
		// the end of transaction.
		IERC20(quoteToken).safeTransferFrom(_msgSender(), address(this), amount);
		IERC20(quoteToken).safeApprove(_ghostBondingAddress, amount);
 
		(uint256 payout,, uint256 index) = IBondDepository(_ghostBondingAddress).deposit(
			bondId,
			amount,
			maxPrice,
			address(this),
			address(this)
		);
 
		if (payout > 0) {
			bondPayouts[tokenId] += payout;
			bondIndexes[tokenId].push(index);
		}
	}
 
	function _claimDiscountedCollateral(uint256 tokenId, uint256[] memory indexes) internal override {
		uint256 payout = INoteKeeper(_ghostBondingAddress).redeem(address(this), indexes, true);
 
		if (payout > 0) {
			(bool shouldAppend,) = _arrayContains(_ghostAddress, collateralTokens[tokenId]);
			if (shouldAppend) {
				collateralTokens[tokenId].push(_ghostAddress);
			}
			collateralBalances[tokenId][_ghostAddress] += payout;
			bondPayouts[tokenId] -= payout;
		}
 
		for (uint256 i = 0; i < indexes.length; i++) {
			uint256 index = _arrayContains(indexes[i], bondIndexes[tokenId]);
			bondIndexes[tokenId][index] = bondIndexes[tokenId][bondIndexes[tokenId].length - 1];
			bondIndexes[tokenId].pop();
		}
	}
 
	function _disperse(address tokenAddress, uint256 tokenId) internal override {
		uint256 balance = disperseBalance[tokenAddress] / totalSupply();
		if (balance > disperseTaken[tokenId][tokenAddress]) {
			uint256 amount = balance - disperseTaken[tokenId][tokenAddress];
			disperseTaken[tokenId][tokenAddress] += amount;
 
			(bool shouldAppend,) = _arrayContains(tokenAddress, collateralTokens[tokenId]);
			if (shouldAppend) {
				collateralTokens[tokenId].push(tokenAddress);
			}
			collateralBalances[tokenId][tokenAddress] += amount;
		}
	}
 
	function _beforeTokenTransfer(
		address from,
        address to,
        uint256 tokenId
	) internal override(ERC721, ERC721Pausable, ERC721Enumerable) {
		ERC721Enumerable._beforeTokenTransfer(from, to, tokenId);
		require(!paused(), "token paused");
	}
 
	function _beforeConsecutiveTokenTransfer(
        address from,
        address to,
        uint256 first,
        uint96 size
    ) internal override(ERC721, ERC721Pausable, ERC721Enumerable) {
		// Due to ERC721Enumerable, while size > 0 tx will be reverted
        ERC721Enumerable._beforeConsecutiveTokenTransfer(from, to, first, size);
    }
}
